import { Injectable } from '@angular/core';
import { Location } from '@angular/common';
import { Router } from '@angular/router';
 
import { MonthlyBillingService } from './monthlyBilling.service';
import { IPaymentExtensionApplication } from './paymentScheme/paymentExtensionApplication.service';
import { IPaymentAssistanceChooseService } from './paymentScheme/paymentAssistanceChoose.service';
 
@Injectable()
export class SecondaryNavigationService {
    public url: string;
 
    private concessionPaths: string[] = [
        '/settings/concession/selectaccount',
        '/settings/concession/selectfuel',
        '/settings/concession/selectcard',
        '/settings/concession/confirmdetails',
        '/settings/concession/apply'
    ];
 
    constructor(
        private router: Router,
        private location: Location,
        private monthlyBillingService: MonthlyBillingService,
        private paymentExtensionApplicationService: IPaymentExtensionApplication,
        private paymentAssistanceChooseService: IPaymentAssistanceChooseService
    ) {}
 
    public get display(): boolean {
        const urlsWithSecondaryNav = [
            '/bills/paymentassistance/welcome',
            '/bills/paymentassistance/extend/select',
            '/bills/paymentassistance/extend/confirm',
            '/bills/paymentassistance/select',
            '/bills/paymentassistance/choose',
            '/bills/paymentassistance/plan',
            '/settings/monthlybilling/services',
            '/settings/monthlybilling/date',
            '/usage/meterreadcheck',
            '/bills/paymentassistance'
        ].concat(this.concessionPaths);
 
        this.url = this.normaliseURLToCompare();
        return urlsWithSecondaryNav.indexOf(this.url.toLowerCase()) > -1;
    }
 
    public get displayBack(): boolean {
        return this.backClick !== null;
    }
 
    public get desktopText(): string {
        const urlToDesktopTextMap = {
            ['/bills/paymentassistance/welcome']: 'Payment extensions',
            ['/bills/paymentassistance/extend/select']: 'Payment extensions',
            ['/bills/paymentassistance/extend/confirm']: 'Payment extensions',
            ['/bills/paymentassistance/extend/success']: 'Payment extensions',
            ['/bills/paymentassistance/select']: 'Payment assistance',
            ['/bills/paymentassistance/choose']: 'Payment assistance',
            ['/bills/paymentassistance/plan']: 'Instalment plans',
            ['/settings/monthlybilling/services']: 'Monthly billing',
            ['/settings/monthlybilling/date']: 'Monthly billing',
            ['/usage/meterreadcheck']: 'Self Service Meter Read'
        };
        this.concessionPaths.forEach((path) => urlToDesktopTextMap[path] = 'Concessions');
 
        this.url = this.normaliseURLToCompare();
        return urlToDesktopTextMap[this.url.toLowerCase()] || '';
    }
 
    public get mobileText(): string {
        // only required if the mobile text is different to the desktop text
        const urlToMobileTextMap = {};
        this.url = this.normaliseURLToCompare();
        return urlToMobileTextMap[this.url.toLowerCase()] || '';
    }
 
    public get backClick() {
        const urlToBackActionMap = {
            ['/settings/concession/selectfuel']: () => this.router.navigate(['/settings/concession']),
            ['/bills/paymentassistance/extend/confirm']: () => this.paymentExtensionApplicationService.backClick(),
            ['/bills/paymentassistance/choose']: () => this.paymentAssistanceChooseService.backClick(),
            ['/bills/paymentassistance/plan']: () => this.location.back(),
            ['/usage/meterreadcheck']: () => this.router.navigate(['/overview']),
            ['/settings/monthlybilling/date']: () => this.router.navigate(['/settings/monthlybilling/services']),
            ['/settings/monthlybilling/services']: () => this.router.navigate([this.monthlyBillingService.getEntryPointUrl()])
        };
        this.concessionPaths.forEach((path) => urlToBackActionMap[path] = () => this.location.back());
 
        this.url = this.normaliseURLToCompare();
        return urlToBackActionMap[this.url.toLowerCase()] || null;
    }
 
    private normaliseURLToCompare() {
        const containsExtensionConfirmUrl = window.location.pathname.includes('/bills/paymentassistance/extend/confirm');
        const containsAssistanceChooseUrl = window.location.pathname.includes('/bills/paymentassistance/choose');
        const selectConcessionFuelUrl = window.location.pathname.includes('/settings/concession/selectfuel');
        const containsAssistancePlanUrl = window.location.pathname.includes('/bills/paymentassistance/plan');
        const containsAssistancePlanSuccessUrl = window.location.pathname.includes('/bills/paymentassistance/plan/success');
 
        if (containsExtensionConfirmUrl) {
            return '/bills/paymentassistance/extend/confirm';
        } else if (containsAssistanceChooseUrl) {
            return '/bills/paymentassistance/choose';
        } else if (containsAssistancePlanSuccessUrl) {
            return '/bills/paymentassistance/plan/success';
        } else if (selectConcessionFuelUrl) {
            return '/settings/concession/selectfuel';
        } else if (containsAssistancePlanUrl) {
            return '/bills/paymentassistance/plan';
        } else {
            return window.location.pathname;
        }
    }
}
