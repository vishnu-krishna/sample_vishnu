import { Injectable } from '@angular/core';
import { Observable } from 'rxjs/Observable';
import { InstalmentPlanFrequency, IPaymentSchemeApi, PaymentArrangementInstalmentPlanOption, PaymentArrangementInstalmentSummary, PaymentArrangementInstalmentSuggestion } from './paymentSchemeApi.service';
 
export abstract class IInstalmentPlanApplication {
    public abstract getInstalmentPlanOptionsByDefault(contractNumber: string): Observable<PaymentArrangementInstalmentPlanOption[]>;
    public abstract getInstalmentPlanOptionsByStartDate(contractNumber: string, startDate: Date): Observable<PaymentArrangementInstalmentPlanOption[]>;
    public abstract getInstalmentMinAmountByStartDate(contractNumber: string, startDate: Date): Observable<PaymentArrangementInstalmentPlanOption[]>;
    public abstract getInstalmentSummary(contractNumber: number, frequency: InstalmentPlanFrequency, startDate: Date, instalmentAmount: number): Observable<PaymentArrangementInstalmentSummary>;
    public abstract submitInstalmentPlan(contractNumber: number, frequency: InstalmentPlanFrequency, startDate: Date, instalmentAmount: number): Observable<PaymentArrangementInstalmentSummary>;
}
 
@Injectable()
export class InstalmentPlanApplicationService implements IInstalmentPlanApplication {
    constructor(
        private api: IPaymentSchemeApi
    ) {}
 
    // is called when "no, set up instalment plan" button is clicked
    public getInstalmentPlanOptionsByDefault(contractNumber: string): Observable<PaymentArrangementInstalmentPlanOption[]> {
        return this.api.getPaymentArrangementInstalmentPlanOptions(contractNumber, true);
    }
 
    // is called when different start date is selected on "set up instalment plan" page
    public getInstalmentPlanOptionsByStartDate(contractNumber: string, startDate: Date): Observable<PaymentArrangementInstalmentPlanOption[]> {
        return this.api.getPaymentArrangementInstalmentPlanOptions(contractNumber, true, startDate);
    }
 
    // is called when different start date is selected on "custom instalment plan" page
    public getInstalmentMinAmountByStartDate(contractNumber: string, startDate: Date): Observable<PaymentArrangementInstalmentPlanOption[]> {
        return this.api.getPaymentArrangementInstalmentPlanOptions(contractNumber, false, startDate);
    }
 
    public getInstalmentSummary(contractNumber: number, frequency: InstalmentPlanFrequency, startDate: Date, instalmentAmount: number): Observable<PaymentArrangementInstalmentSummary> {
        return this.api.getPaymentArrangementInstalmentSummary(contractNumber, frequency, startDate, instalmentAmount);
    }
 
    public submitInstalmentPlan(contractNumber: number, frequency: InstalmentPlanFrequency, startDate: Date, instalmentAmount: number): Observable<PaymentArrangementInstalmentSummary> {
        return this.api.submitPaymentArrangementInstalmentPlan(contractNumber, frequency, startDate, instalmentAmount);
    }
}
