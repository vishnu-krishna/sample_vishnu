import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { AglAuthTokenProvider } from '../../../../shared/repository/aglAuthTokenProvider';
import { ConfigService } from '../../../../shared/service/config.service';
 
@Component({
    selector: 'agl-payment-assistance-welcome',
    template: require('./paymentAssistanceWelcome.component.html'),
    styles: [require('./paymentAssistanceWelcome.component.scss')]
})
export class PaymentAssistanceWelcomeComponent implements OnInit {
 
    public showPage: boolean = false;
 
    constructor(
        private configService: ConfigService,
        private router: Router,
        private aglAuthTokenProvider: AglAuthTokenProvider
    ) {}
 
    public ngOnInit() {
        if (this.aglAuthTokenProvider.hasToken()) {
            this.router.navigate(['/bills/paymentassistance/select']);
        } else {
            this.showPage = true;
        }
    }
 
    public login() {
        this.configService.navigateToLoginWithReturnPath('/bills/paymentassistance/select');
    }
 
    public register() {
        this.configService.navigateToRegisterWithReturnPath('/bills/paymentassistance/select');
    }
}
