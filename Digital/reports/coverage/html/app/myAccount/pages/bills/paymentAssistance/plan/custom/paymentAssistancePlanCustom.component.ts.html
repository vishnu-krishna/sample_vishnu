import * as moment from 'moment';
import * as _ from 'lodash';
 
import { Component, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { Observable } from 'rxjs';
 
import { Now } from '../../../../../../shared/service/now.service';
import { IPaymentExtensionStateService } from '../../extend/services/paymentExtensionState.service';
import { IPaymentExtensionFuelChipService } from '../../extend/eligibility/services/paymentExtensionFuelChip.service';
import { IInstalmentPlanOptionsService, InstalmentCustomPlan, DateOption, InstalmentOption, DateOptions, FrequencyOption } from '../../../../../services/paymentScheme/instalmentPlanOptions.service';
import { FlashMessageType } from '../../../../../maui/flashMessage';
import { InstalmentPlanFrequency, GetPaymentArrangementInstalmentPlanOptionsParams, InstalmentPlanParameters } from '../../../../../services/paymentScheme/paymentSchemeApi.service';
import { MauiFuelChipState, MauiFuelChipFuelContext } from '../../../../../maui/fuelChip';
import { FuelChipData } from '../../extend/eligibility/fuelChipData';
import { IAccountServiceMA } from '../../../../../services/account.service';
import { DateRangeGenerator, DateRange } from './dateRangeGenerator';
import { DateHelper } from '../../../../../../shared/utils/dateHelper';
import { IInstalmentPlanSummaryService } from '../../../../../services/paymentScheme/instalmentPlanSummary.service';
import { FormBuilder, FormGroup, Validators, AbstractControl, ValidationErrors, FormControl } from '@angular/forms';
import { AglValidators } from '../../../../../../shared/validators/aglValidators';
import { timer } from 'rxjs/observable/timer';
 
@Component({
    selector: 'agl-payment-assistance-plan-custom',
    template: require('./paymentAssistancePlanCustom.component.html'),
    styles: [require('./paymentAssistancePlanCustom.component.scss')]
})
export class PaymentAssistancePlanCustomComponent implements OnInit {
    public MauiFuelChipState = MauiFuelChipState;
    public MauiFuelChipFuelContext = MauiFuelChipFuelContext;
    public InstalmentPlanFrequency = InstalmentPlanFrequency;
    public fuelChip: FuelChipData;
    public instalmentOptions: InstalmentOption[];
    public customPageForm: FormGroup;
 
    public isSettingStartDate: boolean;
    public isLoading: boolean;
    public isContinueLoading: boolean;
    public setDateFailed: boolean = false;
    public enableConfirmButton: boolean = false;
    public nextBillIssuedDate: string;
    public subheading: string;
    public frequencyOptions: FrequencyOption[];
    private dateRange: DateRange[];
    public frequencyDates: DateOption[];
 
    public amountErrorMessage: string = '';
    public instalmentMinAmount: number;
    public instalmentMaxAmount: number;
    public minAmountErrorMessage: string;
    public maxAmountErrorMessage: string;
    public amountHasDollarSign: boolean;
    public placeholder: string;
    private toValidate: boolean;
 
    private contractNumber: string;
 
    private dateRangeGenerator: DateRangeGenerator;
    public cancelDestinationUrl: string;
    private currentBillEndDate: Date;
 
    // Post options
    public selectedFrequency: InstalmentPlanFrequency;
    public selectedContractAccountNumber: string;
    public selectedContractNumber: string;
    public getSelectedDate(): moment.Moment { return DateHelper.parseIsoDate(this.frequencyDates.find((d) => d.selected).value); }
    public showInstalmentsError: boolean = false;
    public flashMessageType = FlashMessageType;
 
    constructor(
        private formBuilder: FormBuilder,
        private route: ActivatedRoute,
        private now: Now,
        private accountService: IAccountServiceMA,
        private instalmentPlanOptionsService: IInstalmentPlanOptionsService,
        private instalmentPlanSummaryService: IInstalmentPlanSummaryService,
        private paymentExtensionStateService: IPaymentExtensionStateService,
        private fuelChipService: IPaymentExtensionFuelChipService,
        private router: Router,
    ) {
        this.dateRangeGenerator = new DateRangeGenerator(now);
    }
 
    public ngOnInit() {
        this.route.params.subscribe((routeParams) => {
            this.contractNumber = routeParams.contractNumber;
            if (this.route && this.route.snapshot) {
                this.setCancelLink(this.route.snapshot.queryParams['cancelDestinationUrl']);
            }
 
            if (this.paymentExtensionStateService.getSelectedFuelChip()) {
                this.isLoading = true;
                this.accountService.getAccounts()
                    .finally(() => this.isLoading = false)
                    .subscribe((accounts) => {
                        this.currentBillEndDate = accounts
                            .find((account) => account.accountNumber === routeParams.contractAccountNumber).contracts
                            .find((contract) => contract.contractNumber === routeParams.contractNumber).currentBillEndDate;
 
                        this.fuelChip = this.paymentExtensionStateService.getSelectedFuelChip();
                        this.instalmentOptions = this.instalmentPlanOptionsService.getInitialisedInstalmentOptions();
                        this.prepareCustomInstalmentOptionsView();
                    });
 
            } else {
                this.isLoading = true;
                this.instalmentPlanOptionsService.initCustomInstalmentPlanSession(routeParams.contractAccountNumber, routeParams.contractNumber)
                    .finally(() => this.isLoading = false)
                    .subscribe(
                        (options: InstalmentCustomPlan) => {
                            this.fuelChip = options.fuelChipData;
                            this.instalmentOptions = options.instalmentOptions;
                            this.currentBillEndDate = options.currentBillEndDate;
                            this.prepareCustomInstalmentOptionsView();
                        }, (error) => {
                            // Handle error here
                        });
            }
            this.selectedContractAccountNumber = routeParams.contractAccountNumber;
            this.selectedContractNumber = routeParams.contractNumber;
        });
    }
 
    public selectFrequency(frequency: InstalmentPlanFrequency): void {
        this.selectedFrequency = frequency;
        this.toValidate = true;
        this.customPageForm.get('amount').setValue('');
        this.updateAmountInputData(this.instalmentOptions);
        this.displayDateOptions();
    }
 
    public selectDate(date: string): void {
        this.isSettingStartDate = true;
        this.setDateFailed = false;
        this.toValidate = true;
        this.customPageForm.get('amount').setValue('');
 
        let newlySelectedDate = DateHelper.parseIsoDate(date).toDate();
 
        this.instalmentPlanOptionsService.getInstalmentOptions(this.contractNumber, this.getParams(newlySelectedDate))
            .finally(() => this.isSettingStartDate = false)
            .subscribe(
                (options: InstalmentOption[]) => {
                    let newDates = _.cloneDeep(this.frequencyDates);
 
                    newDates.find((d) => d.selected).selected = false;
                    newDates.find((d) => d.value === date).selected = true;
 
                    this.frequencyDates = newDates;
                    this.updateAmountInputData(options);
                }, (error) => {
                    this.setDateFailed = true;
                });
    }
 
    public getInstalmentSummary(): void {
        if (this.customPageForm.valid) {
            const instalmentAmount = this.customPageForm.get('amount').value;
 
            this.isContinueLoading = true;
            this.showInstalmentsError = false;
            const investmentPlanRouteParameters: InstalmentPlanParameters = new InstalmentPlanParameters();
            investmentPlanRouteParameters.contractNumber = this.selectedContractNumber;
            investmentPlanRouteParameters.frequency = this.selectedFrequency;
            investmentPlanRouteParameters.instalmentAmount = instalmentAmount;
            investmentPlanRouteParameters.startDate = this.getSelectedDate().toDate();
            this.instalmentPlanSummaryService.initInstalmentPlanSummary(investmentPlanRouteParameters)
                .finally(() => this.isContinueLoading = false)
                .subscribe((res) => {
                    const formattedSelectedDate: string = DateHelper.toIsoDateString(this.getSelectedDate());
                    this.router.navigate([`/bills/paymentassistance/plan/confirm/${this.selectedContractAccountNumber}/${this.selectedContractNumber}/${this.selectedFrequency}/${formattedSelectedDate}/${instalmentAmount}`], { queryParamsHandling: 'preserve'});
                },
                    (error) => {
                        this.showInstalmentsError = true;
                    });
        }
        }
 
    public onAmountFocus(event: FocusEvent): void {
        this.toggleDollarSign(true);
        this.amountErrorMessage = '';
        this.enableConfirmButton = false;
    }
 
    public onAmountBlur(event: FocusEvent): void {
        this.toValidate = true;
        let control = this.customPageForm.get('amount');
        control.markAsTouched();
        control.updateValueAndValidity();
    }
 
    private toggleDollarSign(isDollarSign: boolean): void {
        if (isDollarSign) {
            this.amountHasDollarSign = true;
            this.placeholder = '';
        } else {
            this.amountHasDollarSign = false;
            this.placeholder = `Min. $${this.instalmentMinAmount.toFixed(2)}`;
        }
    }
 
    private prepareCustomInstalmentOptionsView(): void {
        this.setSubheading();
        this.setNextBillDate();
        this.setSelectedFrequency();
        this.getFrequencyOptions();
        this.dateRange = this.dateRangeGenerator.generate(this.instalmentOptions);
        this.displayDateOptions();
        this.updateAmountInputData(this.instalmentOptions);
        this.initialiseForm();
    }
    public setCancelLink(url: string): void {
        this.cancelDestinationUrl = url || '/bills';
    }
 
    public cancelClicked(): void {
        this.router.navigate([this.cancelDestinationUrl]);
    }
 
    private initialiseForm(): void {
        this.customPageForm = this.formBuilder.group({
            amount: new FormControl({ value: null, disabled: this.isSettingStartDate }, Validators.compose([
                AglValidators.decimalField,
                this.minAmountValidator.bind(this),
                this.maxAmountValidator.bind(this)
            ]))
        });
 
        this.initFieldValidation('amount', this.amountValidationHandler);
    }
 
    private initFieldValidation(field: string, validationHandler: (c: AbstractControl) => void): void {
        let control = this.customPageForm.get(field);
        control.valueChanges
            .filter(() => this.toValidate)
            .debounce(() => timer(200)).subscribe((val) => {
                validationHandler.call(this, control);
                this.toValidate = false;
            });
    }
 
    private amountValidationHandler(c: AbstractControl): void {
        const validationMessageLookup = {
            amount: `One or more of the characters you've entered are invalid. You may only enter numeric characters and decimals ('.').`,
            minAmount: this.minAmountErrorMessage,
            maxAmount: this.maxAmountErrorMessage
        };
 
        if ((c.touched || c.dirty)) {
            let formatAmount = true;
 
            if (c.errors) {
                const firstError = Object.keys(c.errors)[0]; // just show the 1st error
                this.amountErrorMessage = validationMessageLookup[firstError];
                this.enableConfirmButton = false;
                if (firstError === 'amount') {
                    formatAmount = false;
                }
            } else {
                if (this.isEmpty(c.value)) {
                    this.toggleDollarSign(false);
                    this.enableConfirmButton = false;
                } else {
                    this.enableConfirmButton = true;
                }
                this.amountErrorMessage = '';
            }
 
            if (formatAmount && !this.isEmpty(c.value)) {
                // format decimal places of the value
                const formattedValue = Number(c.value).toFixed(2);
                if (formattedValue !== c.value) {
                    c.patchValue(formattedValue);
                }
            }
        }
    }
 
    private minAmountValidator(control: AbstractControl): ValidationErrors {
        // we treat invalid characters as valid, so it delegates invalid character validation to amount validator
        if (this.isEmpty(control.value) || !AglValidators.isDecimalValid(control.value)) {
            return null;
        }
 
        return (control.value < this.instalmentMinAmount) ? {
            minAmount: control.value
        } : null;
    }
 
    private maxAmountValidator(control: AbstractControl): ValidationErrors {
        // we treat invalid characters as valid, so it delegates invalid character validation to amount validator
        if (this.isEmpty(control.value) || !AglValidators.isDecimalValid(control.value)) {
            return null;
        }
 
        return (control.value > this.instalmentMaxAmount) ? {
            maxAmount: control.value
        } : null;
    }
 
    private setSubheading(): void {
        this.subheading = this.instalmentOptions.length === 1 ?
            'Let us know how much you are able to pay each week.' :
            'Let us know how often and how much you are able to pay.';
    }
 
    private setNextBillDate(): void {
        const nextBillDate = moment(this.currentBillEndDate).startOf('day');
        const today = this.now.date();
        this.nextBillIssuedDate = nextBillDate.isSameOrBefore(today, 'day') ?
            'soon' :
            `on ${nextBillDate.format('DD MMM YYYY')}`;
    }
 
    private setSelectedFrequency(): void {
        this.selectedFrequency = this.instalmentOptions.length > 1 ? InstalmentPlanFrequency.Fortnightly : InstalmentPlanFrequency.Weekly;
    }
 
    private getFrequencyOptions(): void {
        this.frequencyOptions = this.instalmentOptions.map((option) => {
            const frequency = option.frequency.toString();
            const isSelected = option.frequency === this.selectedFrequency;
            return new FrequencyOption(frequency, frequency, isSelected);
        });
    }
 
    private displayDateOptions(): void {
        let frequency = this.dateRange.find((option) => option.frequency === this.selectedFrequency);
        this.frequencyDates = _.cloneDeep(frequency.dates);
    }
 
    private getInstalmentMinAmount(options: InstalmentOption[]): number {
        return options.find((option) => option.frequency === this.selectedFrequency).instalmentMinAmount;
    }
 
    private getInstalmentMaxAmount(options: InstalmentOption[]): number {
        return options.find((option) => option.frequency === this.selectedFrequency).instalmentMaxAmount;
    }
 
    private getParams(selectedDate: Date): GetPaymentArrangementInstalmentPlanOptionsParams {
        return {
            suggestInstalments: false,
            frequency: this.selectedFrequency,
            startDate: selectedDate
        };
    }
 
    private updateAmountInputData(instalmentOptions: InstalmentOption[]): void {
        const currentBillEndDateFormatted = moment(this.currentBillEndDate).startOf('day').format('DD MMM YYYY');
        this.instalmentMinAmount = this.getInstalmentMinAmount(instalmentOptions);
        this.instalmentMaxAmount = this.getInstalmentMaxAmount(instalmentOptions);
        this.minAmountErrorMessage = `In order to pay this bill before your next bill on ${currentBillEndDateFormatted}, you need to pay a minimum of $${this.instalmentMinAmount} per week.`;
        this.maxAmountErrorMessage = `In order to set up an instalment plan online, each of your instalments must be at least $10. The maximum instalment amount you can enter is $${this.instalmentMaxAmount}.`;
        this.toggleDollarSign(false);
    }
 
    private isEmpty(value): boolean {
        if (value === null || value === '') {
            return true;
        }
        return false;
    }
}
