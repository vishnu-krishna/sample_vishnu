import { Injectable } from '@angular/core';
import { HomeProfileViewModel } from './homeProfileViewModel';
import { IHomeProfileMapper } from './homeProfileMapper';
import { IHomeProfileApi, HomeProfileApiModel, HomeProfileApiService, HomeProfileSummaryApiModel } from '../../../services/homeProfile/homeProfileApi.service';
import { Console } from '@angular/core/src/console';
import { Observable } from 'rxjs/Observable';
 
export abstract class IHomeProfileStateService {
    public abstract homeProfile: HomeProfileViewModel;
    public abstract getContractId(): Observable<string>;
    public abstract setContractId(contractId: string): void;
    public abstract getProfile(): Observable<HomeProfileViewModel>;
    public abstract saveProfile(): Observable<boolean>;
    public abstract getStatus(): HomeProfileStatus;
}
 
// This service is still under development, will revisit later after the other payment extension pages are ready.
@Injectable()
export class HomeProfileStateService implements IHomeProfileStateService {
    public homeProfile: HomeProfileViewModel;
    private contractId: string;
 
    constructor(
        private homeProfileApi: IHomeProfileApi,
        private homeProfileMapper: IHomeProfileMapper
    ) {
        this.homeProfile = new HomeProfileViewModel();
    }
 
    public getContractId(): Observable<string> {
        if (this.contractId) {
            return Observable.of(this.contractId);
        }
        return this.homeProfileApi.getProfiles().map((result: HomeProfileSummaryApiModel[]) => {
            this.contractId = (result.length > 0) ? result[0].contractNumber : null;
            return this.contractId;
        });
    }
 
    public setContractId(contractId: string): void {
        this.contractId = contractId;
    }
 
    public getProfile(): Observable<HomeProfileViewModel> {
        this.ensureContractIdIsSet();
        return this.homeProfileApi.getProfile(this.contractId).map((result: HomeProfileApiModel) => {
                this.homeProfile = this.homeProfileMapper.toViewModel(result);
                return this.homeProfile;
            });
    }
 
    public saveProfile(): Observable<boolean> {
        this.ensureContractIdIsSet();
        let profile = this.homeProfileMapper.fromViewModel(this.homeProfile);
        return this.homeProfileApi.saveProfile(this.contractId, profile).map((result) => {
                return true;
            });
    }
 
    public getStatus(): HomeProfileStatus {
        // We are not yet using the 'Complete' status since we do not know at this stage when it is supposed to be complete
        return (!this.homeProfile || JSON.stringify(this.homeProfile) === '{}') ? HomeProfileStatus.NotStarted : HomeProfileStatus.Incomplete;
    }
 
    private ensureContractIdIsSet() {
        if (!this.contractId) {
            return Observable.throw('Contract id not set');
        }
    }
 
}
 
export enum HomeProfileStatus {
    NotStarted,
    Incomplete,
    Complete
}
