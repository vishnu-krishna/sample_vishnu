import { Component, OnInit } from '@angular/core';
import { Observable } from 'rxjs/Observable';
import { ContractViewModel, AccountViewModel } from '../../../../services/account.service';
import { HomeProfileNavigationService, HomeProfilePage } from '../homeProfileNavigation.service';
import { IHomeProfileStateService, HomeProfileStatus } from '../homeProfileState.service';
import { IAccountServiceMA } from '../../../../services/account.service';
import { HomeProfileViewModel } from '../homeProfileViewModel';
import { FlashMessageType } from '../../../../maui/flashMessage';
 
@Component({
    selector: 'agl-home-profile-landing',
    template: require('./homeProfileLanding.component.html'),
    styles: [require('./homeProfileLanding.component.scss')]
})
export class HomeProfileLandingComponent implements OnInit {
    public eligible: boolean;
    public address: string = '';
    public isLoading: boolean = false;
    public isError: boolean = false;
    public flashMessageType = FlashMessageType;
    public page = HomeProfilePage;
 
    constructor(
        private homeProfileNavigationService: HomeProfileNavigationService,
        private accountService: IAccountServiceMA,
        private homeProfileStateService: IHomeProfileStateService
    ) {}
 
    public ngOnInit(): void {
        this.isLoading = true;
        Observable.forkJoin(
                this.accountService.getAccounts(),
                this.homeProfileStateService.getContractId())
            .flatMap(([accounts, contractId]: [AccountViewModel[], string]) => {
                const account: AccountViewModel = (accounts && accounts.length > 0) ? accounts[0] : null;
                this.address = this.getAddress(account);
                return this.getProfile(account, contractId);
            })
            .finally(() => this.isLoading = false)
            .subscribe(
                (homeProfileModel: HomeProfileViewModel) => {
                    this.isError = false;
                },
                (error) => {
                    console.error('Error retrieving home profile');
                    this.isError = true;
                });
    }
 
    public startHomeProfile() {
        this.homeProfileNavigationService.startClick();
    }
 
    public allowStart(): boolean {
        return this.homeProfileStateService.getStatus() === HomeProfileStatus.NotStarted;
    }
 
    public allowEdit(): boolean {
        return this.homeProfileStateService.getStatus() === HomeProfileStatus.Incomplete;
    }
 
    public edit(page: HomeProfilePage) {
        if (this.allowEdit()) {
            this.homeProfileNavigationService.gotoSurveyPage(page);
        }
    }
 
    public getAddress(account: AccountViewModel): string {
        let address = '';
        if (account) {
            // The account.groupedAddress is only populated if there is more than one contract under the account, and if the addresses are the exactly same.
            address = account.groupedAddress
                || (account.contracts.length > 0 ? account.contracts[0].address : address);
        }
        return address;
    }
 
    public getProfile(account: AccountViewModel, contractId: string): Observable<HomeProfileViewModel> {
        if (contractId) {
            if (!this.isContractCurrent(account, contractId)) {
                return Observable.throw('Contract id is not current');
            }
            return this.homeProfileStateService.getProfile();
        } else {
            this.homeProfileStateService.setContractId(account.contracts[0].contractNumber);
            return Observable.of(new HomeProfileViewModel());
        }
    }
 
    public isContractCurrent(account: AccountViewModel, contractId: string): boolean {
        return account && account.contracts.length > 0 && contractId &&
            account.contracts.find((c) => c.contractNumber === contractId) != null;
    }
}
