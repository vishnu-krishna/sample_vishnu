import { Component, Input, OnInit } from '@angular/core';
import { ContractEnergyInsightsModel } from '../../../../services/settings/model/contractEnergyInsightsModel';
import { EnergyInsightsService } from '../../../../services/energyInsights.service';
import { SetupEnergyInsightsRequest } from '../../../../services/settings/model/setupEnergyInsightsRequest';
import { EnergyInsightsEligibility } from '../../../../services/settings/model/energyInsightsEligibility';
import { FlashMessageType } from '../../../../maui/flashMessage';
 
@Component({
    selector: 'agl-settings-manage-energy-insights',
    template: require('./manageEnergyInsights.component.html'),
    styles: [require('./../notifications.component.scss')]
})
 
export class ManageEnergyInsightsComponent implements OnInit {
    @Input() public model: ContractEnergyInsightsModel;
    public FlashMessageType = FlashMessageType;
 
    constructor(
        private energyInsightService: EnergyInsightsService
    ) { }
 
    public emailAddress: string;
    public midBillRequestLoading: boolean = false;
    public endBillRequestLoading: boolean = false;
    public energyInsightsEligibility: EnergyInsightsEligibility;
    public postEnergyInsightsEndFailed: boolean = false;
    public postEnergyInsightsMidFailed: boolean = false;
 
    private contractNumber: string;
 
    ngOnInit(): void {
        this.emailAddress = this.model.email;
        this.contractNumber = this.model.contract.contractNumber;
        this.energyInsightsEligibility = this.model.energyInsightsEligibility;
    }
 
    public setUpMidBillReport(setupMidBill: boolean): void {
        this.midBillRequestLoading = true;
        let energyInsightsRequest = new SetupEnergyInsightsRequest(setupMidBill);
 
        this.energyInsightService.setupEnergyInsights(this.contractNumber, energyInsightsRequest)
            .finally(() => {
                this.midBillRequestLoading = false;
            })
            .subscribe((setupResult) => {
                this.energyInsightsEligibility.subscribedToMidBillEnergyBreakdown = setupMidBill;
                this.postEnergyInsightsMidFailed = false;
            },
            (err) => {
                this.postEnergyInsightsMidFailed = true;
                this.energyInsightsEligibility.subscribedToMidBillEnergyBreakdown  = !setupMidBill;
            });
    }
 
    public setUpEndBillReport(setupEndBill: boolean): void {
        this.endBillRequestLoading = true;
        let energyInsightsRequest = new SetupEnergyInsightsRequest(null, setupEndBill);
 
        this.energyInsightService.setupEnergyInsights(this.contractNumber, energyInsightsRequest)
            .finally(() => {
                this.endBillRequestLoading = false;
            })
            .subscribe((setupResult) => {
                this.energyInsightsEligibility.subscribedToEndBillEnergyBreakdown = setupEndBill;
                this.postEnergyInsightsEndFailed = false;
            },
            (err) => {
                this.postEnergyInsightsEndFailed = true;
                this.energyInsightsEligibility.subscribedToEndBillEnergyBreakdown = !setupEndBill;
            });
    }
 
    public onClickDismissFlashMessage(isMid: boolean) {
        if (isMid) {
            this.postEnergyInsightsMidFailed = false;
        } else {
            this.postEnergyInsightsEndFailed = false;
        }
    }
}
