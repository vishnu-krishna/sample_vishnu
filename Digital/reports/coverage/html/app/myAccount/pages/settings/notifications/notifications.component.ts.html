import { ApiError } from './../../../services/settings/model/apiError';
import { Component, OnInit } from '@angular/core';
import { FeatureFlagService } from '../../../services/featureFlag.service';
import { ConfigService } from '../../../../shared/service/config.service';
import { EnergyInsightsService } from '../../../services/energyInsights.service';
import { EnergyInsightsEligibility } from '../../../services/settings/model/energyInsightsEligibility';
import { FeatureFlagTypes } from '../../../services/featureFlag.constants';
import { Observable } from 'rxjs/Observable';
import { AccountViewModel, ContractViewModel, AccountService } from '../../../services/account.service';
import { ContractEnergyInsightsModel } from '../../../services/settings/model/contractEnergyInsightsModel';
import { BillDeliveryMethod } from '../../../services/settings/model';
import { EmailBillingComponentModel } from '../billing/emailBilling/emailBillingComponentModel';
import { FlashMessageType } from '../../../maui/flashMessage';
 
@Component({
    selector: 'agl-settings-notifications',
    template: require('./notifications.component.html'),
    styles: [require('./notifications.component.scss')]
})
 
export class NotificationsComponent implements OnInit {
 
    public preferencesUrl: string;
    public energyInsightsEligibility: ContractEnergyInsightsModel;
    public FlashMessageType = FlashMessageType;
    public accountList: AccountViewModel[];
    public isEnergyInsightsEligible: boolean = false;
    public isEnergyInsightsApiError: boolean = false;
 
    private isEnergyInsightsEnabled: boolean = false;
    private isEbillLoading: boolean = true;
    private isEnergyInsightsLoading: boolean = true;
    private isAccountValidForEnergyInsights: boolean = true;
 
    constructor(
        private configService: ConfigService,
        private accountService: AccountService,
        private featureFlagService: FeatureFlagService,
        private energyInsightsService: EnergyInsightsService,
    ) {
        this.preferencesUrl = `${this.configService.current.aglSiteCoreWebsiteBaseUrl}/aeo/Home/MyAccount/Update-Details/PreferencesPage`;
    }
 
    public ngOnInit(): void {
        this.getEbillData();
        this.getEnergyInsightsData();
        this.getAccountList();
    }
 
    public isPageLoading(): boolean {
        return this.isEbillLoading || this.isEnergyInsightsLoading;
    }
 
    /*
        We are required to show the energy insights section (with a flash msg) if the API call to get eligibility and subscription fails
        regardless of whether the account is eligible for energy insights or not.
        The account is deemed 'valid' as long as it's single account, single electricity contract.
    */
    public showEnergyInsights(): boolean {
        return this.isEnergyInsightsEnabled
            && this.isAccountValidForEnergyInsights;
    }
 
    public getAccountAddresses(account: AccountViewModel): string[] {
        let accountAddresses = [];
        if (!!account.groupedAddress) {
            accountAddresses = [account.groupedAddress];
        } else {
            account.contracts.forEach((contract) => {
                if (!contract.isRestricted){
                    accountAddresses.push(contract.address);
                }
            });
        }
        return accountAddresses;
    }
 
    public accountHasGas(account: AccountViewModel): boolean {
        return account.contracts.some((contract) => contract.isGas);
    }
 
    public accountHasElectricity(account: AccountViewModel): boolean {
        return account.contracts.some((contract) => contract.isElectricity);
    }
 
    private getAccountList(): void {
        this.accountService.getAccounts()
        .subscribe((accounts) => {
            this.accountList = accounts.filter((acc) => !acc.allContractsAreRestricted);
        },
        (error) => {
            console.log(error);
        });
    }
 
    private getEbillData(): void {
        this.isEbillLoading = false;
        // TODO: Get Bill Delivery Preference Data
    }
 
    private getEnergyInsightsData(): void {
        Observable.forkJoin(
            this.featureFlagService.featureFlagged(FeatureFlagTypes.energyInsightsEnabled),
            this.accountService.getAccounts()
        )
        .subscribe(
            ([energyInsightsFeatureFlag, accountsList]) => {
                this.isEnergyInsightsEnabled = energyInsightsFeatureFlag;
                this.isAccountValidForEnergyInsights = this.energyInsightsService.verifyAccountDetails(accountsList, energyInsightsFeatureFlag);
 
                if (this.isAccountValidForEnergyInsights) {
                    this.energyInsightsService.getContractDetailsAndEligibility(accountsList[0])
                    .finally(() => {
                        this.isEnergyInsightsLoading = false;
                    })
                    .subscribe(
                        (eligibilityList: ContractEnergyInsightsModel[]) => {
                            // for phase 1, we are only expecting 1 contract (elec) to be eligible for energy insights
                            this.energyInsightsEligibility = eligibilityList[0];
                            this.isEnergyInsightsEligible = this.energyInsightsEligibility.energyInsightsEligibility.isEligible;
                        },
                        ((error) => {
                            this.isEnergyInsightsApiError = true;
                        })
                    );
                } else {
                    this.isEnergyInsightsLoading = false;
                }
            },
            ((error) => {
                this.isEnergyInsightsLoading = false;
                this.isEnergyInsightsApiError = true;
            })
        );
    }
}
