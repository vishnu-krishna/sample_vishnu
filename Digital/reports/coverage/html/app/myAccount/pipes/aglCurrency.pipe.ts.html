import { Pipe, PipeTransform } from '@angular/core';
import { CurrencyPipe } from '@angular/common';
 
@Pipe({ name: 'aglCurrency' })
export class AglCurrency implements PipeTransform {
    private currencyPipe = new CurrencyPipe('en-AU');
 
    public transform(value: number) {
        return this.currencyPipe.transform(value, 'AUD', true, '1.2-2');
    }
}
