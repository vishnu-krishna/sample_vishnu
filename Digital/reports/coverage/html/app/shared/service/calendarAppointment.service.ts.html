import { Injectable } from '@angular/core';
 
// This service makes use of a third party service from addevent.com
// https://www.addevent.com
// Future developers do not necessarily need to be added to the addevent portal to do development as everyone uses same the AGL account client id
// Having said that you can log on to the site to see the analytics
// The client id is not displayed obviously on the site. It was obtained from the https://www.addevent.com/add-to-calendar-button page under the "client" parameter
//
// NOTE: On IOS devices, you will need to call the `add` function on a click event of a button or anchor,
// to prevent the default popup blocker from blocking the `window.open` of a 3rd party webpage.
 
@Injectable()
export class CalendarAppointmentService {
    private clientId = 'arohxvEKyzCXGBQPvmGM35825';
 
    public add(icsFileName: string, start: string, timezone: string, title: string, calendarType: CalendarType, description: string): string {
 
        const link: string = this.createLink(icsFileName, start, timezone, title, calendarType, description);
        window.open(link);
 
        return link;
    }
 
    public createLink(icsFileName: string, start: string, timezone: string, title: string, calendarType: CalendarType, description: string) {
        return `https://www.addevent.com/dir/?client=${this.clientId}&calname=${icsFileName}&start=${start}&timezone=${timezone}&title=${title}&service=${calendarType.toString()}&description=${description}`;
    }
}
 
export enum CalendarType {
    Apple = <any> 'apple',
    Google = <any> 'google',
    Outlook = <any> 'outlook',
    OutlookCom = <any> 'outlookcom',
    Yahoo = <any> 'yahoo',
}
